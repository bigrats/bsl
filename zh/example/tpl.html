        </td>
        <td>
            <strong>${item.price}</strong>
        </td>
        <!--[ if(item.change < 0.0d) { ]-->
        <td class="minus">${item.change}</td>
        <td class="minus">${item.ratio}</td>
        <!--[ } else { ]-->
        <td>${item.change}</td>
        <td>${item.ratio}</td>
        <!--[ } ]-->
    </tr>
    <!--[ } ]-->
    </tbody>
</table>
